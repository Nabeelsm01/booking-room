import random responses = { "สวัสดี": "สวัสดีค่ะ! มีอะไรให้ช่วยคะ?", "ชื่ออะไร": "ฉันคือผู้ช่วย AI ของคุณค่ะ", "ทำอะไรได้บ้าง": "ฉันสามารถตอบคำถามพื้นฐานและช่วยคุณได้หลายอย่างค่ะ", "อากาศวันนี้เป็นอย่างไร": "ขออภัยค่ะ ฉันไม่สามารถตรวจสอบสภาพอากาศได้ในขณะนี้", "ลาก่อน": "สวัสดีค่ะ แล้วพบกันใหม่นะคะ!" } def askAI(event): # รับคำถามจากผู้ใช้ user_input = Element("userInput").element.value.lower() if not user_input: Element("response").element.innerText = "กรุณากรอกคำถามค่ะ" else: ai_response = responses.get(user_input, "ขอโทษค่ะ ฉันไม่เข้าใจคำถามนั้น") Element("response").element.innerText = ai_response # ผูกปุ่ม "ถาม AI" กับฟังก์ชัน Element("askButton").element.onclick = askAI
